Overview

India is the world’s second-largest consumer of fertilizers. With domestic production insufficient to meet demand for nitrogenous (Urea), phosphatic (DAP, Rock Phosphate, Phosphoric Acid), and potassic (MOP) nutrients, India imports substantial quantities each year. This analysis examines FY 2016–2024 import trends, source-country concentration, international price volatility, and domestic Urea/DAP production.

## Interactive stacked bar helper
make_fert_girafe <- function(data, title, subtitle, pal = NULL) {
  data <- as.data.table(data)
  data[, tt := paste0("<b>", Country, "</b><br>Year: ", Year,
                      "<br>Qty: ", round(Qty,2), " MT<br>Share: ", Percent, "%")]
  p <- ggplot(data, aes(x = Year, y = Qty, fill = Country)) +
    geom_bar_interactive(aes(tooltip = tt, data_id = Country),
                         stat = "identity", position = "stack", width = 0.75, alpha = 0.9) +
    geom_text(aes(label = ifelse(Percent < 9, "", paste0(Percent, "%"))),
              position = position_stack(vjust = 0.5),
              size = 3, color = "white", fontface = "bold") +
    my_theme +
    scale_y_continuous(labels = comma_format()) +
    labs(x = NULL, y = "Qty (Million Tonnes)", title = title, subtitle = subtitle,
         caption = "Source: Ministry of Commerce, India | Analysis: Pawan")
  if (!is.null(pal)) p <- p + scale_fill_manual(values = pal)
  else               p <- p + scale_fill_brewer(palette = "Set2")
  girafe(ggobj = p, width_svg = 11, height_svg = 6,
         options = list(
           opts_tooltip(css = "background:#fff;color:#333;padding:8px;border-radius:5px;border:1px solid #ddd;font-size:12px;"),
           opts_hover(css = "opacity:0.85;stroke:#333;stroke-width:1.5;"),
           opts_hover_inv(css = "opacity:0.3;"),
           opts_zoom(min = 0.8, max = 3),
           opts_sizing(rescale = TRUE, width = 1)))
}

## Data loading & cleaning helper
load_fert <- function(path) {
  dt <- readRDS(path)
  dt$ImportMonth <- gsub(",","",dt$ImportMonth)
  dt
}

prep_country <- function(dt, keep_countries, factor_levels) {
  dt <- dt[Country != "Total",
           .(Qty = sum(as.numeric(ImportMonth), na.rm=TRUE)), .(Country, Year)]
  dt[!(Country %in% keep_countries), Country := "OTHER COUNTRY"]
  dt <- dt[, .(Qty = sum(Qty)), .(Country, Year)]
  dt[, Qty := Qty/100000]
  dt <- dt[Year > "2015", .(Qty = sum(Qty)), .(Country, Year)]
  dt[, Percent := round(Qty*100/sum(Qty)), by = Year]
  dt[, Country := factor(Country, levels = factor_levels)]
  dt
}
