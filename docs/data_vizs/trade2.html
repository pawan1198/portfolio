Overview

India’s merchandise trade has grown dramatically over the past two decades — exports crossing USD 450 billion in recent years while imports hover around USD 700 billion, producing a persistent and growing trade deficit. This report decomposes the story through HS code level analysis: cereals (HS 10), edible oils (HS 15), fertilizers (HS 31), and petroleum products (HS 27).

## Data cleaning helper
clean_trade <- function(dt, col) {
  dt[, (col) := as.numeric(gsub("[,]","", get(col)))]
  dt[is.na(get(col)), (col) := 0]
  dt[, `:=`(HSCode2=substr(HSCode,1,2),
             HSCode4=substr(HSCode,1,4),
             HSCode6=substr(HSCode,1,6))]
  dt
}

Section 1 · Data Loading & Aggregation

Code 
usd  <- clean_trade(readRDS("/home/pawan/trade/totaltrade/total_exportsusd.rds"),  "Export")
qty  <- clean_trade(readRDS("/home/pawan/trade/totaltrade/total_exportsqty.rds"),  "Export")
musd <- clean_trade(readRDS("/home/pawan/trade/totaltrade/total_importsusd.rds"),  "Import")
mqty <- clean_trade(readRDS("/home/pawan/trade/totaltrade/total_importsqty.rds"),  "Import")

hs2  <- as.data.table(read_excel("/home/pawan/trade/totaltrade/two-digit.xlsx"))
hs2[, HSCode2 := as.integer(HSCode2)]
qty[,  HSCode2 := as.integer(HSCode2)]
mqty[, HSCode2 := as.integer(HSCode2)]
qty  <- merge(qty,  hs2, by="HSCode2", all.x=TRUE)
mqty <- merge(mqty, hs2, by="HSCode2", all.x=TRUE)

## Annual aggregates
exp_usd  <- usd[,  .(Export_USD  = sum(Export,  na.rm=TRUE)/1000), by=Year]
imp_usd  <- musd[, .(Import_USD  = sum(Import,  na.rm=TRUE)/1000), by=Year]
exp_qty  <- qty[,  .(Export_Qty  = sum(Export,  na.rm=TRUE)), by=Year]
imp_qty  <- mqty[, .(Import_Qty  = sum(Import,  na.rm=TRUE)), by=Year]

trade <- Reduce(function(a,b) merge(a,b,by="Year"),
                list(exp_usd, imp_usd, exp_qty, imp_qty))
trade[, `:=`(
  Balance_USD  = Export_USD - Import_USD,
  Volume_USD   = Export_USD + Import_USD,
  Coverage_Pct = round(Export_USD/Import_USD*100, 1),
  Exp_Growth   = round((Export_USD/shift(Export_USD)-1)*100, 1),
  Imp_Growth   = round((Import_USD/shift(Import_USD)-1)*100, 1)
)]

Section 2 · Trade Overview: Exports, Imports & Balance

Code 
trade_long <- melt(trade[, .(Year,Export_USD,Import_USD)],
                   id.vars="Year", variable.name="Flow", value.name="Billion_USD")
trade_long[, tt := paste0(Flow,"\nYear: ",Year,"\n$",round(Billion_USD,1),"B")]

p_flows <- ggplot(trade_long, aes(x=Year, y=Billion_USD, color=Flow,
                                   text=tt, group=Flow)) +
  geom_line_interactive(aes(tooltip=tt, data_id=Flow), linewidth=1.3) +
  geom_point(size=2.5, alpha=0.7) +
  geom_ribbon(data=trade, aes(x=Year, ymin=pmin(Export_USD,Import_USD),
                               ymax=pmax(Export_USD,Import_USD),
                               fill=ifelse(Balance_USD<0,"Deficit","Surplus")),
              inherit.aes=FALSE, alpha=0.15) +
  scale_color_manual(values=c("Export_USD"="#27AE60","Import_USD"="#E74C3C"),
                     labels=c("Export_USD"="Exports","Import_USD"="Imports")) +
  scale_fill_manual(values=c("Deficit"="#FADBD8","Surplus"="#D5F5E3")) +
  scale_y_continuous(labels=dollar_format(suffix="B")) +
  my_theme +
  labs(title="India's Merchandise Trade: Exports & Imports",
       subtitle="Shaded area shows trade deficit (red) | Hover for exact values",
       y="USD Billion", x="Year",
       caption="Source: DGCI&S / Ministry of Commerce | Analysis: Pawan")

girafe(ggobj=p_flows, width_svg=12, height_svg=6,
       options=list(
         opts_tooltip(css="background:#fff;color:#333;padding:8px;border-radius:5px;border:1px solid #ddd;"),
         opts_hover(css="stroke-width:2.5;"),
         opts_hover_inv(css="opacity:0.3;"),

Section 3 · Cereals Export Composition

India is a major global rice exporter and an intermittent wheat exporter. The composition shifts sharply based on domestic supply and export bans/policy.

Code 
cer <- qty[HSCode2 == 10]
cer[, Commodity := fcase(
  HSCode4=="1006", "Rice",
  HSCode4=="1001", "Wheat",
  HSCode4=="1005", "Maize",
  HSCode4=="1008", "Millet/Buckwheat",
  default          = "Other Cereals"
)]
cer_agg <- cer[, .(Qty = sum(Export, na.rm=TRUE)), .(Year, Commodity)]
cer_agg[, Total := sum(Qty), by=Year]
cer_agg[, Share := round(Qty/Total*100)]
cer_agg[, tt    := paste0("<b>",Commodity,"</b><br>Year: ",Year,
                           "<br>Qty: ",format(Qty,big.mark=",")," K Tonnes",
                           "<br>Share: ",Share,"%")]

p_cer <- ggplot(cer_agg, aes(x=Year, y=Qty, fill=Commodity)) +
  geom_bar_interactive(aes(tooltip=tt, data_id=Commodity),
                        stat="identity", position="stack", width=0.75, alpha=0.9) +

  scale_y_continuous(labels=comma_format()) +

  labs(title="India's Cereal Exports by Commodity",

cer <- qty[HSCode2 == 10]
cer[, Commodity := fcase(
  HSCode4=="1006", "Rice",
  HSCode4=="1001", "Wheat",
  HSCode4=="1005", "Maize",
  HSCode4=="1008", "Millet/Buckwheat",
  default          = "Other Cereals"
)]
cer_agg <- cer[, .(Qty = sum(Export, na.rm=TRUE)), .(Year, Commodity)]
cer_agg[, Total := sum(Qty), by=Year]
cer_agg[, Share := round(Qty/Total*100)]
cer_agg[, tt    := paste0("<b>",Commodity,"</b><br>Year: ",Year,
                           "<br>Qty: ",format(Qty,big.mark=",")," K Tonnes",
                           "<br>Share: ",Share,"%")]

p_cer <- ggplot(cer_agg, aes(x=Year, y=Qty, fill=Commodity)) +
  geom_bar_interactive(aes(tooltip=tt, data_id=Commodity),
                        stat="identity", position="stack", width=0.75, alpha=0.9) +
  geom_text(aes(label=ifelse(Share<7,"",paste0(Share,"%"))),
            position=position_stack(vjust=0.5), size=2.8, color="white", fontface="bold") +
  scale_fill_brewer(palette="Set1") +
  scale_y_continuous(labels=comma_format()) +
  my_theme +
  labs(title="India's Cereal Exports by Commodity",
       subtitle="Rice dominates; wheat exports surge in surplus years",
       y="Quantity (000 Tonnes)", x="Year",
       caption="Source: DGCI&S | HS Chapter 10 | Analysis: Pawan")

girafe(ggobj=p_cer, width_svg=12, height_svg=6,
       options=list(
         opts_tooltip(css="background:#fff;color:#333;padding:8px;border-radius:5px;border:1px solid #ddd;"),
         opts_hover(css="opacity:0.85;stroke-width:1.5;"),
         opts_hover_inv(css="opacity:0.3;"),
         opts_sizing(rescale=TRUE,width=1)))
