1 Introduction to Descriptive Statistics

Descriptive statistics summarize and describe the main features of a dataset. They are the first step in any data analysis — before we run a single regression or hypothesis test, we must understand what our data looks like.

In R, we have powerful tools for this. Let’s load our packages and create a realistic dataset:

Code 
library(tidyverse)
library(knitr)

# Simulate Labour Force Survey data
set.seed(2025)
n <- 1000

lfs_data <- data.frame(
  age       = round(rnorm(n, 35, 12)) |> pmax(15) |> pmin(70),
  wage      = exp(rnorm(n, log(400), 0.7)) |> round(),
  edu_years = sample(0:20, n, replace = TRUE, 
                     prob = dpois(0:20, 8) + 0.01),
  gender    = sample(c("Male","Female"), n, replace = TRUE, prob=c(.6,.4)),
  state     = sample(c("Haryana","UP","Bihar","Maharashtra","Karnataka"), 
                     n, replace = TRUE),
  sector    = sample(c("Agriculture","Industry","Services"), n, 
                     replace = TRUE, prob = c(.4,.2,.4))
)

2 Measures of Central Tendency

The three main measures of central tendency each tell a different story:

Code 
wage_stats <- lfs_data |>
  summarise(
    Mean   = mean(wage),
    Median = median(wage),
    Mode   = as.numeric(names(sort(table(round(wage,-2)), decreasing=TRUE)[1])),
    SD     = sd(wage),
    IQR    = IQR(wage),
    Min    = min(wage),
    Max    = max(wage),
    Skewness = (mean(wage) - median(wage)) / sd(wage) * 3  # Pearson's approx
  ) |>
  pivot_longer(everything(), names_to = "Statistic", values_to = "Value") |>
  mutate(Value = round(Value, 2))

kable(wage_stats, caption = "Descriptive Statistics for Daily Wage (₹)")

2.1 Why the Mean ≠ Median for Wages

Code 
ggplot(lfs_data, aes(x = wage)) +
  geom_histogram(bins = 50, fill = "#93c5fd", alpha = 0.8, color = "white") +
  geom_vline(aes(xintercept = mean(wage), linetype = "Mean"), 
             color = "#2563eb", linewidth = 1.2) +
  geom_vline(aes(xintercept = median(wage), linetype = "Median"), 
             color = "#e11d48", linewidth = 1.2) +
  scale_linetype_manual(name = "", values = c("Mean"="solid","Median"="dashed")) +
  scale_x_continuous(labels = scales::label_number(prefix = "₹")) +
  labs(title = "Daily Wage Distribution",
       subtitle = "Right skew pulls the mean above the median",
       x = "Daily Wage (₹)", y = "Count") +
  theme_minimal(base_size = 11) +
  theme(legend.position = "top", panel.grid.minor = element_blank())

Key insight: For right-skewed income distributions, the median is almost always a better measure of “typical” than the mean. A few high earners inflate the mean.

3 Measures of Dispersion

Code 
# Boxplot comparison across sectors
ggplot(lfs_data, aes(x = sector, y = wage, fill = sector)) +
  geom_boxplot(alpha = 0.8, outlier.shape = 21, outlier.alpha = 0.5,
               outlier.size = 1.5) +
  stat_summary(fun = mean, geom = "point", shape = 23, 
               size = 4, fill = "white", color = "black") +
  scale_fill_manual(values = c("#2563eb","#7c3aed","#0d9488")) +
  scale_y_continuous(labels = scales::label_number(prefix = "₹")) +
  labs(title = "Wage Dispersion by Sector",
       subtitle = "Diamonds = means · Boxes = IQR · Whiskers = 1.5×IQR",
       x = NULL, y = "Daily Wage (₹)") +
  theme_minimal(base_size = 11) +
  theme(legend.position = "none", panel.grid.minor = element_blank())
Code 
dispersion_by_sector <- lfs_data |>
  group_by(sector) |>
  summarise(
    n      = n(),
    Mean   = round(mean(wage)),
    SD     = round(sd(wage)),
    `CV (%)` = round(sd(wage)/mean(wage)*100, 1),
    Median = round(median(wage)),
    IQR    = round(IQR(wage)),
    .groups = "drop"
  )

kable(dispersion_by_sector,

4.2 Detecting Outliers

Code 
# IQR method for outlier detection
wage_bounds <- lfs_data |>
  summarise(
    Q1  = quantile(wage, 0.25),
    Q3  = quantile(wage, 0.75),
    IQR = IQR(wage),
    lower = Q1 - 1.5 * IQR,
    upper = Q3 + 1.5 * IQR
  )

lfs_outliers <- lfs_data |>
  mutate(is_outlier = wage < wage_bounds$lower | wage > wage_bounds$upper)

cat("Outliers detected (IQR method):", sum(lfs_outliers$is_outlier), "\n")
Outliers detected (IQR method): 54
Code 
cat("That's", round(mean(lfs_outliers$is_outlier)*100, 1), "% of observations\n")
That's 5.4 % of observations
